Feature Selenium Playwright

‚úÖ Selenium Advantages

  • Mature ecosystem (15+ years)
  • Large community support
  • Extensive documentation
  • Enterprise adoption
  • Grid support for scaling
  • Wide language support

‚ö° Playwright Advantages

  • Modern architecture
  • Better performance
  • Built-in features (screenshots, videos)
  • Better debugging tools
  • Native mobile testing
  • Network manipulation
Recommendation for frndlyTV: Start with Selenium for web testing due to team familiarity and enterprise support, but evaluate Playwright for new test suites, especially for mobile web testing and complex interactions.

Recommendation: Use k6 for modern CI/CD integration and developer-friendly scripting. JMeter for complex protocols and if GUI-based test creation is preferred by QA team.

Page Object Implementation

public class LoginPage extends BasePage { // Page elements @FindBy(id = "email") private WebElement emailField; @FindBy(id = "password") private WebElement passwordField; @FindBy(xpath = "//button[@type='submit']") private WebElement loginButton; @FindBy(className = "error-message") private WebElement errorMessage; // Page actions public HomePage login(String email, String password) { emailField.clear(); emailField.sendKeys(email); passwordField.clear(); passwordField.sendKeys(password); loginButton.click(); waitForPageLoad(); return new HomePage(driver); } public boolean isLoginErrorDisplayed() { return isElementPresent(By.className("error-message")); } public String getErrorMessage() { return errorMessage.getText(); } }

Test Implementation

public class LoginTests extends BaseTest { @Test(priority = 1) public void testValidLogin() { LoginPage loginPage = new LoginPage(driver); loginPage.navigateTo(ConfigReader.getProperty("login.url")); HomePage homePage = loginPage.login( TestDataProvider.getValidEmail(), TestDataProvider.getValidPassword() ); Assert.assertTrue(homePage.isUserLoggedIn(), "User should be logged in"); Assert.assertTrue(homePage.isChannelGuideVisible(), "Channel guide should be visible"); } @Test(priority = 2) public void testInvalidLogin() { LoginPage loginPage = new LoginPage(driver); loginPage.navigateTo(ConfigReader.getProperty("login.url")); loginPage.login("invalid@email.com", "wrongpassword"); Assert.assertTrue(loginPage.isLoginErrorDisplayed(), "Error message should be displayed"); Assert.assertTrue(loginPage.getErrorMessage().contains("Invalid credentials")); } @Test(priority = 3) public void testEmptyFieldValidation() { LoginPage loginPage = new LoginPage(driver); loginPage.navigateTo(ConfigReader.getProperty("login.url")); loginPage.login("", ""); Assert.assertTrue(loginPage.isLoginErrorDisplayed(), "Validation error should be shown"); } }

Jenkins Pipeline Configuration

pipeline { agent any parameters { choice(name: 'BROWSER', choices: ['chrome', 'firefox', 'edge'], description: 'Browser selection') choice(name: 'ENVIRONMENT', choices: ['staging', 'qa', 'production'], description: 'Environment') choice(name: 'TEST_SUITE', choices: ['smoke', 'regression', 'full'], description: 'Test suite') } stages { stage('Checkout') { steps { git branch: 'main', url: 'https://github.com/frndly/automation-tests' } } stage('Build') { steps { sh 'mvn clean compile' } } stage('Test Execution') { parallel { stage('UI Tests') { steps { sh "mvn test -Dbrowser=${params.BROWSER} -Denvironment=${params.ENVIRONMENT} -Dsuite=${params.TEST_SUITE}" } } stage('API Tests') { steps { sh 'mvn test -Dtest=ApiTestSuite' } } } } stage('Performance Tests') { when { expression { params.TEST_SUITE == 'full' } } steps { sh 'k6 run performance-tests/load-test.js' } } } post { always { publishTestResults testResultsPattern: 'target/surefire-reports/*.xml' publishHTML([ allowMissing: false, alwaysLinkToLastBuild: true, keepAll: true, reportDir: 'target/surefire-reports', reportFiles: 'index.html', reportName: 'Test Report' ]) // Archive screenshots on failure archiveArtifacts artifacts: 'target/screenshots/**/*.png', allowEmptyArchive: true // Send notifications emailext ( subject: "Test Results: \${env.JOB_NAME} - \${env.BUILD_NUMBER}", body: "Build: \${env.BUILD_URL}", to: "\${env.CHANGE_AUTHOR_EMAIL}" ) } failure { slackSend channel: '#qa-alerts', color: 'danger', message: "Test Failed: \${env.JOB_NAME} - \${env.BUILD_NUMBER}" } success { slackSend channel: '#qa-alerts', color: 'good', message: "Tests Passed: \${env.JOB_NAME} - \${env.BUILD_NUMBER}" } } }
